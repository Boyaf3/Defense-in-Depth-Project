changeto context CTX-MAIN
conf t
hostname ASA-CTX-MAIN

! ======== Interfaces / zones ========

! OUTSIDE – to MANAMA router (209.10.0.0/29)
interface Ethernet1
 nameif outside
 security-level 0
 ip address 209.10.0.2 255.255.255.248 standby 209.10.0.3
 no shutdown

! INSIDE – to IPS (10.10.1.0/29)
interface Ethernet0
 nameif inside
 security-level 100
 ip address 10.10.1.2 255.255.255.248 standby 10.10.1.3
 no shutdown

! DMZ – VLAN 100 (10.10.100.0/24)
interface Ethernet2
 nameif dmz
 security-level 50
 ip address 10.10.100.1 255.255.255.0 standby 10.10.100.2
 no shutdown

! ======== Routing ========

! Default route to MANAMA router
route outside 0.0.0.0 0.0.0.0 209.10.0.1

! All Manama VLANs 10.10.x.x reachable via IPS 10.10.1.1

route inside 10.10.10.0 255.255.255.0 10.10.1.1
route inside 10.10.20.0 255.255.255.0 10.10.1.1
route inside 10.10.30.0 255.255.255.0 10.10.1.1
route inside 10.10.40.0 255.255.255.0 10.10.1.1
route inside 10.10.50.0 255.255.255.0 10.10.1.1
policy-map global_policy
 class inspection_default
inspect icmp

! ======== NAT (PAT + Static) ========

! PAT for inside networks (10.10.10/20/30/40/50)
object network INSIDE-NET
 subnet 10.10.0.0 255.255.0.0
 nat (inside,outside) dynamic interface

! Static NAT for DMZ web server
object network DMZ-WEB
 host 10.10.100.10
 nat (dmz,outside) static 209.10.0.4

! Static NAT for DMZ email server
object network DMZ-MAIL
 host 10.10.100.20
 nat (dmz,outside) static 209.10.0.5

! ======== Network object-groups ========

object-group network DMZ-SERVERS
 network-object host 10.10.100.10
 network-object host 10.10.100.20

object-group network MANAMA-INTERNAL-NETS
 network-object 10.10.10.0 255.255.255.0
 network-object 10.10.20.0 255.255.255.0
 network-object 10.10.30.0 255.255.255.0
 network-object 10.10.40.0 255.255.255.0
 network-object 10.10.50.0 255.255.255.0

! ======== Service objects + service object-group ========

object-group service DMZ-SERVICES
 service-object tcp eq 80
 service-object tcp eq 443
 service-object tcp eq 25
 service-object icmp



! ======== Outside ACL (using object-groups) ========

! Allow only HTTP/HTTPS/SMTP to the public DMZ IPs
access-list OUTSIDE-IN extended permit object-group DMZ-SERVICES any object-group DMZ-SERVERS

! Everything else from outside is denied by default.

access-group OUTSIDE-IN in interface outside

! ======== Management from Management VLAN (10.10.50.0/24) ========

http server enable
http 10.10.50.0 255.255.255.0 inside

ssh 10.10.50.0 255.255.255.0 inside
username admin password cisco123 privilege 15
aaa authentication ssh console LOCAL

end
write memory
