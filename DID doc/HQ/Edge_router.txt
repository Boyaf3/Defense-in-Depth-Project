enable
conf t
hostname MANAMA-RTR
no ip domain-lookup
ip cef

crypto isakmp policy 10
 encr aes
 hash sha
 authentication pre-share
 group 2
 lifetime 86400

crypto isakmp key DMVPN-KEY address 0.0.0.0 0.0.0.0

crypto ipsec transform-set TS-DMVPN esp-aes esp-sha-hmac
 mode transport

crypto ipsec profile DMVPN-PROFILE
 set transform-set TS-DMVPN
! ===== UNDERLAY WAN LINKS =====
interface s1/0
 description Manama <-> Dubai
 ip address 201.10.0.5 255.255.255.252
 no shutdown

interface s1/1
 description Manama <-> Doha
 ip address 201.10.0.1 255.255.255.252
 no shutdown

///////////////
int e0/0
 ip add 209.10.0.1 255.255.255.248
 no shut 
! Loopback used as DMVPN NBMA
interface loopback0
 ip address 10.255.0.1 255.255.255.255
 no shutdown

! ===== DMVPN CLOUD 1 – PRIMARY HUB (Tunnel0) =====
interface Tunnel0
 description DMVPN PRIMARY HUB (Manama)
 ip address 172.31.0.1 255.255.255.0
 ip mtu 1400
 ip tcp adjust-mss 1360
 tunnel source loopback0
 tunnel mode gre multipoint
 tunnel key 100
 ip nhrp network-id 100
 ip nhrp authentication DMVPN
 ip nhrp map multicast dynamic
 ip nhrp redirect
 tunnel protection ipsec profile DMVPN-PROFILE
 no shutdown
 ip ospf 10 area 0
 ip ospf network broadcast
 ip ospf cost 10

! ===== DMVPN CLOUD 2 – BACKUP (Manama SPOKE to Doha) =====
interface Tunnel1
 description DMVPN BACKUP (to Doha HUB)
 ip address 172.31.1.2 255.255.255.0
 ip mtu 1400
 ip tcp adjust-mss 1360
 tunnel source loopback0
 tunnel mode gre multipoint
 tunnel key 200
 ip nhrp network-id 200
 ip nhrp authentication DMVPN
 ip nhrp nhs 172.31.1.1
 ip nhrp map 172.31.1.1 10.255.0.2
 ip nhrp map multicast 10.255.0.2
 ip nhrp shortcut
 tunnel protection ipsec profile DMVPN-PROFILE
 no shutdown
 ip ospf 10 area 0
 ip ospf network point-to-multipoint
 ip ospf cost 100

! ===== OSPF =====
router ospf 10
 router-id 10.255.0.1
 network 201.10.0.0 0.0.0.3 area 0      ! Manama-Doha 201.10.0.0/30
 network 201.10.0.4 0.0.0.3 area 0      ! Manama-Dubai 201.10.0.4/30
 network 10.255.0.1 0.0.0.0 area 0      ! loopback (DMVPN source)
 network 209.10.0.0 0.0.0.15 area 0
exit
ip route 10.10.100.50 255.255.255.255 209.10.0.6
end
wr mem
